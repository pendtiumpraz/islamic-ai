```typescript
  sessionId       String    @unique @default(cuid())

  endedAt         DateTime?
  
  // Messages
  messages        LessonChatMessage[]
  
  // Progress
  phase           LessonPhase @default(TEACHING)
  assessmentScore Int?
  passed          Boolean?
  
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
  
  @@index([userId, lessonId])
}

model LessonChatMessage {
  id              String    @id @default(cuid())
  chatId          String
  chat            LessonChat @relation(fields: [chatId], references: [id])
  
  role            MessageRole  // USER, ASSISTANT, SYSTEM

  structuredData  Json?                   // AI's parsed response
  responseType    ResponseType?           // teaching, assessment, etc.

  questionId      String?

  // Metadata
  tokenCount      Int?
  processingTime  Int?      // ms
  
  createdAt       DateTime  @default(now())
  
  @@index([chatId, createdAt])
```

```prisma
// ==================== LEARN MODE ====================

// (Already defined in main schema)

// ==================== HAFIDZ MODE ====================

model HafalanCategory {
  id              String    @id @default(cuid())
  type            HafalanType  // QURAN, HADITS, MATAN
  
  // For QURAN
  surahNumber     Int?
  surahName       String?
  juzNumber       Int?
  
  // For HADITS
  hadithCollection String?   // bukhari, muslim, etc.
  hadithBook      String?
  
  // For MATAN
  matanSlug       String?
  
  items           HafalanItem[]
}

model HafalanItem {
  id              String    @id @default(cuid())
  categoryId      String
  category        HafalanCategory @relation(fields: [categoryId], references: [id])
  
  // Content
  arabicText      String    @db.Text
  transliteration String?   @db.Text
  translation     Json      // {en, id}
  
  // Metadata
  reference       String    // QS 2:255, HR Bukhari 1, etc.
  audioUrl        String?
  order           Int
  
  // For Quran
  ayahNumber      Int?
  
  // For Hadits
  hadithNumber    Int?
  grade           String?   // shahih, hasan, dhaif
  narrator        String?
  
  // Relations
  progress        HafalanProgress[]
  submissions     HafalanSubmission[]
}

model HafalanProgress {
  id              String    @id @default(cuid())
  userId          String
  user            User      @relation(fields: [userId], references: [id])
  itemId          String
  item            HafalanItem @relation(fields: [itemId], references: [id])
  
  status          HafalanStatus @default(NOT_STARTED)
  bestScore       Int       @default(0)
  attempts        Int       @default(0)
  
  // Spaced repetition
  nextReviewDate  DateTime?
  interval        Int       @default(1)
  easeFactor      Float     @default(2.5)
  
  lastPracticed   DateTime?
  masteredAt      DateTime?
  
  @@unique([userId, itemId])
}

model HafalanSubmission {
  id              String    @id @default(cuid())
  userId          String
  user            User      @relation(fields: [userId], references: [id])
  itemId          String
  item            HafalanItem @relation(fields: [itemId], references: [id])
  
  // Audio
  audioUrl        String?
  duration        Int?
  
  // AI Evaluation
  transcription   String?   @db.Text
  overallScore    Int
  accuracyScore   Int
  tajwidScore     Int?      // Quran only
  makhrajScore    Int
  fluencyScore    Int
  
  mistakes        Json?
  feedback        String?   @db.Text
  
  createdAt       DateTime  @default(now())
}

// ==================== WRITE MODE ====================

model WritingProject {
  id              String    @id @default(cuid())
  userId          String
  user            User      @relation(fields: [userId], references: [id])
  tenantId        String?
  
  title           String
  type            WritingType
  category        String
  targetAudience  String?
  language        Language  @default(ID)
  
  status          WritingStatus @default(DRAFT)
  
  // Content
  synopsis        String?   @db.Text
  outline         Json?
  
  // Stats
  totalWords      Int       @default(0)
  totalPages      Int       @default(0)
  
  chapters        Chapter[]
  dalilUsed       DalilReference[]
  
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
  publishedAt     DateTime?
}

model Chapter {
  id              String    @id @default(cuid())
  projectId       String
  project         WritingProject @relation(fields: [projectId], references: [id])
  
  title           String
  content         String    @db.Text
  order           Int
  wordCount       Int       @default(0)
  
  status          ChapterStatus @default(DRAFT)
  
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
}

model DalilReference {
  id              String    @id @default(cuid())
  projectId       String
  project         WritingProject @relation(fields: [projectId], references: [id])
  
  type            DalilType   // QURAN, HADITS, ATSAR, SIRAH
  reference       String      // QS 2:255, HR Bukhari 1234
  arabicText      String      @db.Text
  translation     String?     @db.Text
  source          String?
  grade           String?     // For hadits
  
  usedInChapter   Int?
  
  createdAt       DateTime  @default(now())
}

model SavedDalil {
  id              String    @id @default(cuid())
  userId          String
  user            User      @relation(fields: [userId], references: [id])
  
  type            DalilType
  reference       String
  arabicText      String    @db.Text
  translation     String?   @db.Text
  source          String?
  notes           String?
  tags            String[]
  
  createdAt       DateTime  @default(now())
}

model SavedQuote {
  id              String    @id @default(cuid())
  userId          String
  user            User      @relation(fields: [userId], references: [id])
  
  type            QuoteType   // SALAF, ULAMA, SIRAH, HIKMAH
  text            String      @db.Text
  arabicText      String?     @db.Text
  attribution     String      // Who said it
  source          String?
  tags            String[]
  
  createdAt       DateTime  @default(now())
}
```
